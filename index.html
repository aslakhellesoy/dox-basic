Parse source code to produce documentation

exports.parse = function(source, options){
    options = options || {};
    var title = options.title || 'Documentation';
    var dox = require('dox');
    var obj = dox.parseComments(source, options);

    return highlight(template({title: 'dox-basic', comments:obj}));
}
